15#[register_rule(("Constant", 9001))]
16fn apply_eval_constant(expr: &Expr, _: &SymbolTable) -> ApplicationResult {
17    if let Expr::Atomic(_, Atom::Literal(_)) = expr {
18        return Err(ApplicationError::RuleNotApplicable);
19    }
20    eval_constant(expr)
21        .map(|c| Reduction::pure(Expr::Atomic(Metadata::new(), Atom::Literal(c))))
22        .ok_or(ApplicationError::RuleNotApplicable)
23}

25/// Simplify an expression to a constant if possible
26/// Returns:
27/// `None` if the expression cannot be simplified to a constant (e.g. if it contains a variable)
28/// `Some(Const)` if the expression can be simplified to a constant
29pub fn eval_constant(expr: &Expr) -> Option<Lit> {
30    match expr {
31        Expr::Atomic(_, Atom::Literal(c)) => Some(c.clone()),
32        Expr::Atomic(_, Atom::Reference(_c)) => None,
33        Expr::Abs(_, e) => un_op::<i32, i32>(|a| a.abs(), e).map(Lit::Int),
34        Expr::Eq(_, a, b) => bin_op::<i32, bool>(|a, b| a == b, a, b)
35            .or_else(|| bin_op::<bool, bool>(|a, b| a == b, a, b))
36            .map(Lit::Bool),
37        Expr::Neq(_, a, b) => bin_op::<i32, bool>(|a, b| a != b, a, b).map(Lit::Bool),
38        Expr::Lt(_, a, b) => bin_op::<i32, bool>(|a, b| a < b, a, b).map(Lit::Bool),
39        Expr::Gt(_, a, b) => bin_op::<i32, bool>(|a, b| a > b, a, b).map(Lit::Bool),
40        Expr::Leq(_, a, b) => bin_op::<i32, bool>(|a, b| a <= b, a, b).map(Lit::Bool),
41        Expr::Geq(_, a, b) => bin_op::<i32, bool>(|a, b| a >= b, a, b).map(Lit::Bool),
42
43        Expr::Not(_, expr) => un_op::<bool, bool>(|e| !e, expr).map(Lit::Bool),
44
45        Expr::And(_, exprs) => vec_op::<bool, bool>(|e| e.iter().all(|&e| e), exprs).map(Lit::Bool),
46        // this is done elsewhere instead - root should return a new root with a literal inside it,
47        // not a literal
48        Expr::Root(_, _) => None,
49        Expr::Or(_, exprs) => vec_op::<bool, bool>(|e| e.iter().any(|&e| e), exprs).map(Lit::Bool),
50        Expr::Imply(_, box1, box2) => {
51            let a: &Atom = (&**box1).try_into().ok()?;
52            let b: &Atom = (&**box2).try_into().ok()?;
53
54            let a: bool = a.try_into().ok()?;
55            let b: bool = b.try_into().ok()?;
56
57            if a {
58                // true -> b ~> b
59                Some(Lit::Bool(b))
60            } else {
61                // false -> b ~> true
62                Some(Lit::Bool(true))
63            }
64        }
65
66        Expr::Sum(_, exprs) => vec_op::<i32, i32>(|e| e.iter().sum(), exprs).map(Lit::Int),
67        Expr::Product(_, exprs) => vec_op::<i32, i32>(|e| e.iter().product(), exprs).map(Lit::Int),
68
69        Expr::FlatIneq(_, a, b, c) => {
70            let a: i32 = a.try_into().ok()?;
71            let b: i32 = b.try_into().ok()?;
72            let c: i32 = c.try_into().ok()?;
73
74            Some(Lit::Bool(a <= b + c))
75        }
76
77        Expr::FlatSumGeq(_, exprs, a) => {
78            let sum = exprs.iter().try_fold(0, |acc, atom: &Atom| {
79                let n: i32 = atom.try_into().ok()?;
80                let acc = acc + n;
81                Some(acc)
82            })?;
83
84            Some(Lit::Bool(sum >= a.try_into().ok()?))
85        }
86        Expr::FlatSumLeq(_, exprs, a) => {
87            let sum = exprs.iter().try_fold(0, |acc, atom: &Atom| {
88                let n: i32 = atom.try_into().ok()?;
89                let acc = acc + n;
90                Some(acc)
91            })?;
92
93            Some(Lit::Bool(sum >= a.try_into().ok()?))
94        }
95        Expr::Min(_, exprs) => {
96            opt_vec_op::<i32, i32>(|e| e.iter().min().copied(), exprs).map(Lit::Int)
97        }
98        Expr::Max(_, exprs) => {
99            opt_vec_op::<i32, i32>(|e| e.iter().max().copied(), exprs).map(Lit::Int)
100        }
101        Expr::UnsafeDiv(_, a, b) | Expr::SafeDiv(_, a, b) => {
102            if unwrap_expr::<i32>(b)? == 0 {
103                return None;
104            }
105            bin_op::<i32, i32>(|a, b| ((a as f32) / (b as f32)).floor() as i32, a, b).map(Lit::Int)
106        }
107        Expr::UnsafeMod(_, a, b) | Expr::SafeMod(_, a, b) => {
108            if unwrap_expr::<i32>(b)? == 0 {
109                return None;
110            }
111            bin_op::<i32, i32>(|a, b| a - b * (a as f32 / b as f32).floor() as i32, a, b)
112                .map(Lit::Int)
113        }
114        Expr::MinionDivEqUndefZero(_, a, b, c) => {
115            // div always rounds down
116            let a: i32 = a.try_into().ok()?;
117            let b: i32 = b.try_into().ok()?;
118            let c: i32 = c.try_into().ok()?;
119
120            if b == 0 {
121                return None;
122            }
123
124            let a = a as f32;
125            let b = b as f32;
126            let div: i32 = (a / b).floor() as i32;
127            Some(Lit::Bool(div == c))
128        }
129        Expr::Bubble(_, a, b) => bin_op::<bool, bool>(|a, b| a && b, a, b).map(Lit::Bool),
130
131        Expr::MinionReify(_, a, b) => {
132            let result = eval_constant(a)?;
133
134            let result: bool = result.try_into().ok()?;
135            let b: bool = b.try_into().ok()?;
136
137            Some(Lit::Bool(b == result))
138        }
139
140        Expr::MinionReifyImply(_, a, b) => {
141            let result = eval_constant(a)?;
142
143            let result: bool = result.try_into().ok()?;
144            let b: bool = b.try_into().ok()?;
145
146            if b {
147                Some(Lit::Bool(result))
148            } else {
149                Some(Lit::Bool(true))
150            }
151        }
152        Expr::MinionModuloEqUndefZero(_, a, b, c) => {
153            // From Savile Row. Same semantics as division.
154            //
155            //   a - (b * floor(a/b))
156            //
157            // We don't use % as it has the same semantics as /. We don't use / as we want to round
158            // down instead, not towards zero.
159
160            let a: i32 = a.try_into().ok()?;
161            let b: i32 = b.try_into().ok()?;
162            let c: i32 = c.try_into().ok()?;
163
164            if b == 0 {
165                return None;
166            }
167
168            let modulo = a - b * (a as f32 / b as f32).floor() as i32;
169            Some(Lit::Bool(modulo == c))
170        }
171
172        Expr::MinionPow(_, a, b, c) => {
173            // only available for positive a b c
174
175            let a: i32 = a.try_into().ok()?;
176            let b: i32 = b.try_into().ok()?;
177            let c: i32 = c.try_into().ok()?;
178
179            if a <= 0 {
180                return None;
181            }
182
183            if b <= 0 {
184                return None;
185            }
186
187            if c <= 0 {
188                return None;
189            }
190
191            Some(Lit::Bool(a ^ b == c))
192        }
193
194        Expr::AllDiff(_, es) => {
195            let mut lits: HashSet<Lit> = HashSet::new();
196            for expr in es {
197                let Expr::Atomic(_, Atom::Literal(x)) = expr else {
198                    return None;
199                };
200                if lits.contains(x) {
201                    return Some(Lit::Bool(false));
202                } else {
203                    lits.insert(x.clone());
204                }
205            }
206            Some(Lit::Bool(true))
207        }
208        Expr::FlatWatchedLiteral(_, _, _) => None,
209        Expr::AuxDeclaration(_, _, _) => None,
210        Expr::Neg(_, a) => {
211            let a: &Atom = a.try_into().ok()?;
212            let a: i32 = a.try_into().ok()?;
213            Some(Lit::Int(-a))
214        }
215        Expr::Minus(_, a, b) => {
216            let a: &Atom = a.try_into().ok()?;
217            let a: i32 = a.try_into().ok()?;
218
219            let b: &Atom = b.try_into().ok()?;
220            let b: i32 = b.try_into().ok()?;
221
222            Some(Lit::Int(a - b))
223        }
224        Expr::FlatMinusEq(_, a, b) => {
225            let a: i32 = a.try_into().ok()?;
226            let b: i32 = b.try_into().ok()?;
227            Some(Lit::Bool(a == -b))
228        }
229        Expr::FlatProductEq(_, a, b, c) => {
230            let a: i32 = a.try_into().ok()?;
231            let b: i32 = b.try_into().ok()?;
232            let c: i32 = c.try_into().ok()?;
233            Some(Lit::Bool(a * b == c))
234        }
235        Expr::FlatWeightedSumLeq(_, cs, vs, total) => {
236            let cs: Vec<i32> = cs
237                .iter()
238                .map(|x| TryInto::<i32>::try_into(x).ok())
239                .collect::<Option<Vec<i32>>>()?;
240            let vs: Vec<i32> = vs
241                .iter()
242                .map(|x| TryInto::<i32>::try_into(x).ok())
243                .collect::<Option<Vec<i32>>>()?;
244            let total: i32 = total.try_into().ok()?;
245
246            let sum: i32 = izip!(cs, vs).fold(0, |acc, (c, v)| acc + (c * v));
247
248            Some(Lit::Bool(sum <= total))
249        }
250
251        Expr::FlatWeightedSumGeq(_, cs, vs, total) => {
252            let cs: Vec<i32> = cs
253                .iter()
254                .map(|x| TryInto::<i32>::try_into(x).ok())
255                .collect::<Option<Vec<i32>>>()?;
256            let vs: Vec<i32> = vs
257                .iter()
258                .map(|x| TryInto::<i32>::try_into(x).ok())
259                .collect::<Option<Vec<i32>>>()?;
260            let total: i32 = total.try_into().ok()?;
261
262            let sum: i32 = izip!(cs, vs).fold(0, |acc, (c, v)| acc + (c * v));
263
264            Some(Lit::Bool(sum >= total))
265        }
266        Expr::FlatAbsEq(_, x, y) => {
267            let x: i32 = x.try_into().ok()?;
268            let y: i32 = y.try_into().ok()?;
269
270            Some(Lit::Bool(x == y.abs()))
271        }
272
273        Expr::UnsafePow(_, a, b) | Expr::SafePow(_, a, b) => {
274            let a: &Atom = a.try_into().ok()?;
275            let a: i32 = a.try_into().ok()?;
276
277            let b: &Atom = b.try_into().ok()?;
278            let b: i32 = b.try_into().ok()?;
279
280            if (a != 0 || b != 0) && b >= 0 {
281                Some(Lit::Int(a ^ b))
282            } else {
283                None
284            }
285        }
286    }
287}

289/// Evaluate the root expression.
290///
291/// This returns either Expr::Root([true]) or Expr::Root([false]).
292#[register_rule(("Constant", 9001))]
293fn eval_root(expr: &Expr, _: &SymbolTable) -> ApplicationResult {
294    // this is its own rule not part of apply_eval_constant, because root should return a new root
295    // with a literal inside it, not just a literal
296
297    let Expr::Root(_, exprs) = expr else {
298        return Err(RuleNotApplicable);
299    };
300
301    match exprs.len() {
302        0 => Ok(Reduction::pure(Expr::Root(
303            Metadata::new(),
304            vec![true.into()],
305        ))),
306        1 => Err(RuleNotApplicable),
307        _ => {
308            let lit =
309                vec_op::<bool, bool>(|e| e.iter().all(|&e| e), exprs).ok_or(RuleNotApplicable)?;
310
311            Ok(Reduction::pure(Expr::Root(
312                Metadata::new(),
313                vec![lit.into()],
314            )))
315        }
316    }
317}

319fn un_op<T, A>(f: fn(T) -> A, a: &Expr) -> Option<A>
320where
321    T: TryFrom<Lit>,
322{
323    let a = unwrap_expr::<T>(a)?;
324    Some(f(a))
325}
326
327fn bin_op<T, A>(f: fn(T, T) -> A, a: &Expr, b: &Expr) -> Option<A>
328where
329    T: TryFrom<Lit>,
330{
331    let a = unwrap_expr::<T>(a)?;
332    let b = unwrap_expr::<T>(b)?;
333    Some(f(a, b))
334}

336#[allow(dead_code)]
337fn tern_op<T, A>(f: fn(T, T, T) -> A, a: &Expr, b: &Expr, c: &Expr) -> Option<A>
338where
339    T: TryFrom<Lit>,
340{
341    let a = unwrap_expr::<T>(a)?;
342    let b = unwrap_expr::<T>(b)?;
343    let c = unwrap_expr::<T>(c)?;
344    Some(f(a, b, c))
345}

347fn vec_op<T, A>(f: fn(Vec<T>) -> A, a: &[Expr]) -> Option<A>
348where
349    T: TryFrom<Lit>,
350{
351    let a = a.iter().map(unwrap_expr).collect::<Option<Vec<T>>>()?;
352    Some(f(a))
353}
354
355fn opt_vec_op<T, A>(f: fn(Vec<T>) -> Option<A>, a: &[Expr]) -> Option<A>
356where
357    T: TryFrom<Lit>,
358{
359    let a = a.iter().map(unwrap_expr).collect::<Option<Vec<T>>>()?;
360    f(a)
361}
362
363#[allow(dead_code)]
364fn flat_op<T, A>(f: fn(Vec<T>, T) -> A, a: &[Expr], b: &Expr) -> Option<A>
365where
366    T: TryFrom<Lit>,
367{
368    let a = a.iter().map(unwrap_expr).collect::<Option<Vec<T>>>()?;
369    let b = unwrap_expr::<T>(b)?;
370    Some(f(a, b))
371}
372
373fn unwrap_expr<T: TryFrom<Lit>>(expr: &Expr) -> Option<T> {
374    let c = eval_constant(expr)?;
375    TryInto::<T>::try_into(c).ok()
376}
