15#[register_rule(("Constant", 9001))]
16fn apply_eval_constant(expr: &Expr, _: &SymbolTable) -> ApplicationResult {
17    if let Expr::Atomic(_, Atom::Literal(_)) = expr {
18        return Err(ApplicationError::RuleNotApplicable);
19    }
20    eval_constant(expr)
21        .map(|c| Reduction::pure(Expr::Atomic(Metadata::new(), Atom::Literal(c))))
22        .ok_or(ApplicationError::RuleNotApplicable)
23}

25/// Simplify an expression to a constant if possible
26/// Returns:
27/// `None` if the expression cannot be simplified to a constant (e.g. if it contains a variable)
28/// `Some(Const)` if the expression can be simplified to a constant
29pub fn eval_constant(expr: &Expr) -> Option<Lit> {
30    match expr {
31        // `fromSolution()` pulls a literal value from last found solution
32        Expr::FromSolution(_, _) => None,
33        // Same as Expr::Root, we should not replace the dominance relation with a constant
34        Expr::DominanceRelation(_, _) => None,
35        Expr::Atomic(_, Atom::Literal(c)) => Some(c.clone()),
36        Expr::Atomic(_, Atom::Reference(_c)) => None,
37        Expr::Abs(_, e) => un_op::<i32, i32>(|a| a.abs(), e).map(Lit::Int),
38        Expr::Eq(_, a, b) => bin_op::<i32, bool>(|a, b| a == b, a, b)
39            .or_else(|| bin_op::<bool, bool>(|a, b| a == b, a, b))
40            .map(Lit::Bool),
41        Expr::Neq(_, a, b) => bin_op::<i32, bool>(|a, b| a != b, a, b).map(Lit::Bool),
42        Expr::Lt(_, a, b) => bin_op::<i32, bool>(|a, b| a < b, a, b).map(Lit::Bool),
43        Expr::Gt(_, a, b) => bin_op::<i32, bool>(|a, b| a > b, a, b).map(Lit::Bool),
44        Expr::Leq(_, a, b) => bin_op::<i32, bool>(|a, b| a <= b, a, b).map(Lit::Bool),
45        Expr::Geq(_, a, b) => bin_op::<i32, bool>(|a, b| a >= b, a, b).map(Lit::Bool),
46
47        Expr::Not(_, expr) => un_op::<bool, bool>(|e| !e, expr).map(Lit::Bool),
48
49        Expr::And(_, exprs) => vec_op::<bool, bool>(|e| e.iter().all(|&e| e), exprs).map(Lit::Bool),
50        // this is done elsewhere instead - root should return a new root with a literal inside it,
51        // not a literal
52        Expr::Root(_, _) => None,
53        Expr::Or(_, exprs) => vec_op::<bool, bool>(|e| e.iter().any(|&e| e), exprs).map(Lit::Bool),
54        Expr::Imply(_, box1, box2) => {
55            let a: &Atom = (&**box1).try_into().ok()?;
56            let b: &Atom = (&**box2).try_into().ok()?;
57
58            let a: bool = a.try_into().ok()?;
59            let b: bool = b.try_into().ok()?;
60
61            if a {
62                // true -> b ~> b
63                Some(Lit::Bool(b))
64            } else {
65                // false -> b ~> true
66                Some(Lit::Bool(true))
67            }
68        }
69
70        Expr::Sum(_, exprs) => vec_op::<i32, i32>(|e| e.iter().sum(), exprs).map(Lit::Int),
71        Expr::Product(_, exprs) => vec_op::<i32, i32>(|e| e.iter().product(), exprs).map(Lit::Int),
72
73        Expr::FlatIneq(_, a, b, c) => {
74            let a: i32 = a.try_into().ok()?;
75            let b: i32 = b.try_into().ok()?;
76            let c: i32 = c.try_into().ok()?;
77
78            Some(Lit::Bool(a <= b + c))
79        }
80
81        Expr::FlatSumGeq(_, exprs, a) => {
82            let sum = exprs.iter().try_fold(0, |acc, atom: &Atom| {
83                let n: i32 = atom.try_into().ok()?;
84                let acc = acc + n;
85                Some(acc)
86            })?;
87
88            Some(Lit::Bool(sum >= a.try_into().ok()?))
89        }
90        Expr::FlatSumLeq(_, exprs, a) => {
91            let sum = exprs.iter().try_fold(0, |acc, atom: &Atom| {
92                let n: i32 = atom.try_into().ok()?;
93                let acc = acc + n;
94                Some(acc)
95            })?;
96
97            Some(Lit::Bool(sum >= a.try_into().ok()?))
98        }
99        Expr::Min(_, exprs) => {
100            opt_vec_op::<i32, i32>(|e| e.iter().min().copied(), exprs).map(Lit::Int)
101        }
102        Expr::Max(_, exprs) => {
103            opt_vec_op::<i32, i32>(|e| e.iter().max().copied(), exprs).map(Lit::Int)
104        }
105        Expr::UnsafeDiv(_, a, b) | Expr::SafeDiv(_, a, b) => {
106            if unwrap_expr::<i32>(b)? == 0 {
107                return None;
108            }
109            bin_op::<i32, i32>(|a, b| ((a as f32) / (b as f32)).floor() as i32, a, b).map(Lit::Int)
110        }
111        Expr::UnsafeMod(_, a, b) | Expr::SafeMod(_, a, b) => {
112            if unwrap_expr::<i32>(b)? == 0 {
113                return None;
114            }
115            bin_op::<i32, i32>(|a, b| a - b * (a as f32 / b as f32).floor() as i32, a, b)
116                .map(Lit::Int)
117        }
118        Expr::MinionDivEqUndefZero(_, a, b, c) => {
119            // div always rounds down
120            let a: i32 = a.try_into().ok()?;
121            let b: i32 = b.try_into().ok()?;
122            let c: i32 = c.try_into().ok()?;
123
124            if b == 0 {
125                return None;
126            }
127
128            let a = a as f32;
129            let b = b as f32;
130            let div: i32 = (a / b).floor() as i32;
131            Some(Lit::Bool(div == c))
132        }
133        Expr::Bubble(_, a, b) => bin_op::<bool, bool>(|a, b| a && b, a, b).map(Lit::Bool),
134
135        Expr::MinionReify(_, a, b) => {
136            let result = eval_constant(a)?;
137
138            let result: bool = result.try_into().ok()?;
139            let b: bool = b.try_into().ok()?;
140
141            Some(Lit::Bool(b == result))
142        }
143
144        Expr::MinionReifyImply(_, a, b) => {
145            let result = eval_constant(a)?;
146
147            let result: bool = result.try_into().ok()?;
148            let b: bool = b.try_into().ok()?;
149
150            if b {
151                Some(Lit::Bool(result))
152            } else {
153                Some(Lit::Bool(true))
154            }
155        }
156        Expr::MinionModuloEqUndefZero(_, a, b, c) => {
157            // From Savile Row. Same semantics as division.
158            //
159            //   a - (b * floor(a/b))
160            //
161            // We don't use % as it has the same semantics as /. We don't use / as we want to round
162            // down instead, not towards zero.
163
164            let a: i32 = a.try_into().ok()?;
165            let b: i32 = b.try_into().ok()?;
166            let c: i32 = c.try_into().ok()?;
167
168            if b == 0 {
169                return None;
170            }
171
172            let modulo = a - b * (a as f32 / b as f32).floor() as i32;
173            Some(Lit::Bool(modulo == c))
174        }
175
176        Expr::MinionPow(_, a, b, c) => {
177            // only available for positive a b c
178
179            let a: i32 = a.try_into().ok()?;
180            let b: i32 = b.try_into().ok()?;
181            let c: i32 = c.try_into().ok()?;
182
183            if a <= 0 {
184                return None;
185            }
186
187            if b <= 0 {
188                return None;
189            }
190
191            if c <= 0 {
192                return None;
193            }
194
195            Some(Lit::Bool(a ^ b == c))
196        }
197
198        Expr::AllDiff(_, es) => {
199            let mut lits: HashSet<Lit> = HashSet::new();
200            for expr in es {
201                let Expr::Atomic(_, Atom::Literal(x)) = expr else {
202                    return None;
203                };
204                if lits.contains(x) {
205                    return Some(Lit::Bool(false));
206                } else {
207                    lits.insert(x.clone());
208                }
209            }
210            Some(Lit::Bool(true))
211        }
212        Expr::FlatWatchedLiteral(_, _, _) => None,
213        Expr::AuxDeclaration(_, _, _) => None,
214        Expr::Neg(_, a) => {
215            let a: &Atom = a.try_into().ok()?;
216            let a: i32 = a.try_into().ok()?;
217            Some(Lit::Int(-a))
218        }
219        Expr::Minus(_, a, b) => {
220            let a: &Atom = a.try_into().ok()?;
221            let a: i32 = a.try_into().ok()?;
222
223            let b: &Atom = b.try_into().ok()?;
224            let b: i32 = b.try_into().ok()?;
225
226            Some(Lit::Int(a - b))
227        }
228        Expr::FlatMinusEq(_, a, b) => {
229            let a: i32 = a.try_into().ok()?;
230            let b: i32 = b.try_into().ok()?;
231            Some(Lit::Bool(a == -b))
232        }
233        Expr::FlatProductEq(_, a, b, c) => {
234            let a: i32 = a.try_into().ok()?;
235            let b: i32 = b.try_into().ok()?;
236            let c: i32 = c.try_into().ok()?;
237            Some(Lit::Bool(a * b == c))
238        }
239        Expr::FlatWeightedSumLeq(_, cs, vs, total) => {
240            let cs: Vec<i32> = cs
241                .iter()
242                .map(|x| TryInto::<i32>::try_into(x).ok())
243                .collect::<Option<Vec<i32>>>()?;
244            let vs: Vec<i32> = vs
245                .iter()
246                .map(|x| TryInto::<i32>::try_into(x).ok())
247                .collect::<Option<Vec<i32>>>()?;
248            let total: i32 = total.try_into().ok()?;
249
250            let sum: i32 = izip!(cs, vs).fold(0, |acc, (c, v)| acc + (c * v));
251
252            Some(Lit::Bool(sum <= total))
253        }
254
255        Expr::FlatWeightedSumGeq(_, cs, vs, total) => {
256            let cs: Vec<i32> = cs
257                .iter()
258                .map(|x| TryInto::<i32>::try_into(x).ok())
259                .collect::<Option<Vec<i32>>>()?;
260            let vs: Vec<i32> = vs
261                .iter()
262                .map(|x| TryInto::<i32>::try_into(x).ok())
263                .collect::<Option<Vec<i32>>>()?;
264            let total: i32 = total.try_into().ok()?;
265
266            let sum: i32 = izip!(cs, vs).fold(0, |acc, (c, v)| acc + (c * v));
267
268            Some(Lit::Bool(sum >= total))
269        }
270        Expr::FlatAbsEq(_, x, y) => {
271            let x: i32 = x.try_into().ok()?;
272            let y: i32 = y.try_into().ok()?;
273
274            Some(Lit::Bool(x == y.abs()))
275        }
276
277        Expr::UnsafePow(_, a, b) | Expr::SafePow(_, a, b) => {
278            let a: &Atom = a.try_into().ok()?;
279            let a: i32 = a.try_into().ok()?;
280
281            let b: &Atom = b.try_into().ok()?;
282            let b: i32 = b.try_into().ok()?;
283
284            if (a != 0 || b != 0) && b >= 0 {
285                Some(Lit::Int(a ^ b))
286            } else {
287                None
288            }
289        }
290        Expr::Scope(_, _) => None,
291    }
292}

294/// Evaluate the root expression.
295///
296/// This returns either Expr::Root([true]) or Expr::Root([false]).
297#[register_rule(("Constant", 9001))]
298fn eval_root(expr: &Expr, _: &SymbolTable) -> ApplicationResult {
299    // this is its own rule not part of apply_eval_constant, because root should return a new root
300    // with a literal inside it, not just a literal
301
302    let Expr::Root(_, exprs) = expr else {
303        return Err(RuleNotApplicable);
304    };
305
306    match exprs.len() {
307        0 => Ok(Reduction::pure(Expr::Root(
308            Metadata::new(),
309            vec![true.into()],
310        ))),
311        1 => Err(RuleNotApplicable),
312        _ => {
313            let lit =
314                vec_op::<bool, bool>(|e| e.iter().all(|&e| e), exprs).ok_or(RuleNotApplicable)?;
315
316            Ok(Reduction::pure(Expr::Root(
317                Metadata::new(),
318                vec![lit.into()],
319            )))
320        }
321    }
322}

324fn un_op<T, A>(f: fn(T) -> A, a: &Expr) -> Option<A>
325where
326    T: TryFrom<Lit>,
327{
328    let a = unwrap_expr::<T>(a)?;
329    Some(f(a))
330}
331
332fn bin_op<T, A>(f: fn(T, T) -> A, a: &Expr, b: &Expr) -> Option<A>
333where
334    T: TryFrom<Lit>,
335{
336    let a = unwrap_expr::<T>(a)?;
337    let b = unwrap_expr::<T>(b)?;
338    Some(f(a, b))
339}

341#[allow(dead_code)]
342fn tern_op<T, A>(f: fn(T, T, T) -> A, a: &Expr, b: &Expr, c: &Expr) -> Option<A>
343where
344    T: TryFrom<Lit>,
345{
346    let a = unwrap_expr::<T>(a)?;
347    let b = unwrap_expr::<T>(b)?;
348    let c = unwrap_expr::<T>(c)?;
349    Some(f(a, b, c))
350}

352fn vec_op<T, A>(f: fn(Vec<T>) -> A, a: &[Expr]) -> Option<A>
353where
354    T: TryFrom<Lit>,
355{
356    let a = a.iter().map(unwrap_expr).collect::<Option<Vec<T>>>()?;
357    Some(f(a))
358}

360fn opt_vec_op<T, A>(f: fn(Vec<T>) -> Option<A>, a: &[Expr]) -> Option<A>
361where
362    T: TryFrom<Lit>,
363{
364    let a = a.iter().map(unwrap_expr).collect::<Option<Vec<T>>>()?;
365    f(a)
366}

368#[allow(dead_code)]
369fn flat_op<T, A>(f: fn(Vec<T>, T) -> A, a: &[Expr], b: &Expr) -> Option<A>
370where
371    T: TryFrom<Lit>,
372{
373    let a = a.iter().map(unwrap_expr).collect::<Option<Vec<T>>>()?;
374    let b = unwrap_expr::<T>(b)?;
375    Some(f(a, b))
376}
377
378fn unwrap_expr<T: TryFrom<Lit>>(expr: &Expr) -> Option<T> {
379    let c = eval_constant(expr)?;
380    TryInto::<T>::try_into(c).ok()
381}
