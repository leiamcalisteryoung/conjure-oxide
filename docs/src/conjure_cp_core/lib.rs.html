21/// Creates a [`Domain::Int`](ast::Domain::Int).
22///
23/// # Examples
24/// ```
25/// use conjure_cp_core::{domain_int,range,ast::Domain};
26///
27/// let a = 2*10;
28/// assert_eq!(domain_int!(1..5,a+2,), Domain::Int(vec![range!(1..5),range!(a+2)]));
29/// assert_eq!(domain_int!(), Domain::Int(vec![]));
30/// ```
31#[macro_export]
32macro_rules! domain_int {
33    () => {$crate::ast::Domain::Int(vec![])};
34
35    // when parsing expressions, rust groups 1..2 into a single token tree, (1..2)
36    // however, we want it to be three seperate token trees [1,..,2] for parsing.
37    // use defile to turn it back into 3 token trees
38    ($($e:expr),+ $(,)?) => {::defile::defile! { $crate::ast::Domain::Int(vec![$($crate::range!(@$e)),+]) } };
39}

41/// Creates a [`Range`](ast::Range).
42///
43/// # Examples
44///
45/// ```
46/// use conjure_cp_core::{range,ast::Range};
47///
48/// let a = 2*10;
49/// assert_eq!(range!(..a),Range::UnboundedL(a));
50/// assert_eq!(range!(2*5..),Range::UnboundedR(10));
51/// assert_eq!(range!(..10),Range::UnboundedL(10));
52/// assert_eq!(range!(1..5),Range::Bounded(1,5));
53/// assert_eq!(range!(Some(10).unwrap()),Range::Single(10));
54/// ```
55#[macro_export]
56macro_rules! range {
57    // decl macros have no lookahead, hence this nonsense with pushdown automata.
58
59    // @hasLowerbound: have atleast one token of the lower bound on the stack
60    (@hasLowerBound [$($lower:tt)+] -> ..) => {$crate::ast::Range::UnboundedR($crate::as_expr!($($lower)+))};
61    (@hasLowerBound [$($lower:tt)+] -> .. $($tail:tt)+) => {$crate::ast::Range::Bounded($crate::as_expr!($($lower)+),$crate::as_expr!($($tail)+))};
62    (@hasLowerBound [$($lower:tt)+] -> $b:tt $($tail:tt)*) => {range!(@hasLowerBound [$($lower)+ $b] -> $($tail)*)};
63    (@hasLowerBound [$($lower:tt)+] ->) => {$crate::ast::Range::Single($crate::as_expr!($($lower)+))};
64
65    // initial tokens
66    (.. $($a:tt)+) => {$crate::ast::Range::UnboundedL($crate::as_expr!($($a)+))};
67
68    ($a:tt $($tail:tt)*) => {range!(@hasLowerBound [$a] -> $($tail)*)};
69
70}
