18pub fn parse_essence_file_native(
19    path: &str,
20    filename: &str,
21    extension: &str,
22    context: Arc<RwLock<Context<'static>>>,
23) -> Result<Model, EssenceParseError> {
24    let (tree, source_code) = get_tree(path, filename, extension);
25
26    let mut model = Model::new(context);
27    let root_node = tree.root_node();
28    for statement in named_children(&root_node) {
29        match statement.kind() {
30            "single_line_comment" => {}
31            "find_statement_list" => {
32                let var_hashmap = parse_find_statement(statement, &source_code);
33                for (name, decision_variable) in var_hashmap {
34                    model
35                        .as_submodel_mut()
36                        .symbols_mut()
37                        .insert(Rc::new(Declaration::new_var(name, decision_variable)));
38                }
39            }
40            "constraint_list" => {
41                let mut constraint_vec: Vec<Expression> = Vec::new();
42                for constraint in named_children(&statement) {
43                    if constraint.kind() != "single_line_comment" {
44                        constraint_vec.push(parse_constraint(constraint, &source_code));
45                    }
46                }
47                model.as_submodel_mut().add_constraints(constraint_vec);
48            }
49            "e_prime_label" => {}
50            "letting_statement_list" => {
51                let letting_vars = parse_letting_statement(statement, &source_code);
52                model.as_submodel_mut().symbols_mut().extend(letting_vars);
53            }
54            _ => {
55                let kind = statement.kind();
56                return Err(EssenceParseError::ParseError(Error::Parse(
57                    format!("Unrecognized top level statement kind: {kind}").to_owned(),
58                )));
59            }
60        }
61    }
62    Ok(model)
63}

65fn get_tree(path: &str, filename: &str, extension: &str) -> (Tree, String) {
66    let source_code = fs::read_to_string(format!("{path}/{filename}.{extension}"))
67        .expect("Failed to read the source code file");
68    let mut parser = Parser::new();
69    parser.set_language(&LANGUAGE.into()).unwrap();
70    (
71        parser
72            .parse(source_code.clone(), None)
73            .expect("Failed to parse"),
74        source_code,
75    )
76}

78fn parse_find_statement(find_statement_list: Node, source_code: &str) -> BTreeMap<Name, Domain> {
79    let mut vars = BTreeMap::new();
80
81    for find_statement in named_children(&find_statement_list) {
82        let mut temp_symbols = BTreeSet::new();
83
84        let variable_list = find_statement
85            .named_child(0)
86            .expect("No variable list found");
87        for variable in named_children(&variable_list) {
88            let variable_name = &source_code[variable.start_byte()..variable.end_byte()];
89            temp_symbols.insert(variable_name);
90        }
91
92        let domain = find_statement.named_child(1).expect("No domain found");
93        let domain = parse_domain(domain, source_code);
94
95        for name in temp_symbols {
96            vars.insert(Name::UserName(String::from(name)), domain.clone());
97        }
98    }
99    vars
100}

102fn parse_domain(domain: Node, source_code: &str) -> Domain {
103    let domain = domain.child(0).expect("No domain");
104    match domain.kind() {
105        "bool_domain" => Domain::BoolDomain,
106        "int_domain" => parse_int_domain(domain, source_code),
107        "variable" => {
108            let variable_name = &source_code[domain.start_byte()..domain.end_byte()];
109            Domain::DomainReference(Name::UserName(String::from(variable_name)))
110        }
111        _ => panic!("Not bool or int domain"),
112    }
113}

115fn parse_int_domain(int_domain: Node, source_code: &str) -> Domain {
116    if int_domain.child_count() == 1 {
117        Domain::IntDomain(vec![Range::Bounded(std::i32::MIN, std::i32::MAX)])
118    } else {
119        let mut ranges: Vec<Range<i32>> = Vec::new();
120        let range_list = int_domain
121            .named_child(0)
122            .expect("No range list found (expression ranges not supported yet");
123        for int_range in named_children(&range_list) {
124            match int_range.kind() {
125                "integer" => {
126                    let integer_value = &source_code[int_range.start_byte()..int_range.end_byte()]
127                        .parse::<i32>()
128                        .unwrap();
129                    ranges.push(Range::Single(*integer_value));
130                }
131                "int_range" => {
132                    let lower_bound: Option<i32>;
133                    let upper_bound: Option<i32>;
134                    let range_component = int_range.child(0).expect("Error with integer range");
135                    match range_component.kind() {
136                        "expression" => {
137                            lower_bound = Some(
138                                source_code
139                                    [range_component.start_byte()..range_component.end_byte()]
140                                    .parse::<i32>()
141                                    .unwrap(),
142                            );
143
144                            if let Some(range_component) = range_component.next_named_sibling() {
145                                upper_bound = Some(
146                                    source_code
147                                        [range_component.start_byte()..range_component.end_byte()]
148                                        .parse::<i32>()
149                                        .unwrap(),
150                                );
151                            } else {
152                                upper_bound = None;
153                            }
154                        }
155                        ".." => {
156                            lower_bound = None;
157                            let range_component = range_component
158                                .next_sibling()
159                                .expect("Error with integer range");
160                            upper_bound = Some(
161                                source_code
162                                    [range_component.start_byte()..range_component.end_byte()]
163                                    .parse::<i32>()
164                                    .unwrap(),
165                            );
166                        }
167                        _ => panic!("unsupported int range type"),
168                    }
169
170                    match (lower_bound, upper_bound) {
171                        (Some(lb), Some(ub)) => ranges.push(Range::Bounded(lb, ub)),
172                        (Some(lb), None) => ranges.push(Range::Bounded(lb, std::i32::MAX)),
173                        (None, Some(ub)) => ranges.push(Range::Bounded(std::i32::MIN, ub)),
174                        _ => panic!("Unsupported int range type"),
175                    }
176                }
177                _ => panic!("unsupported int range type"),
178            }
179        }
180        Domain::IntDomain(ranges)
181    }
182}

184fn parse_letting_statement(letting_statement_list: Node, source_code: &str) -> SymbolTable {
185    let mut symbol_table = SymbolTable::new();
186
187    for letting_statement in named_children(&letting_statement_list) {
188        let mut temp_symbols = BTreeSet::new();
189
190        let variable_list = letting_statement.child(0).expect("No variable list found");
191        for variable in named_children(&variable_list) {
192            let variable_name = &source_code[variable.start_byte()..variable.end_byte()];
193            temp_symbols.insert(variable_name);
194        }
195
196        let expr_or_domain = letting_statement
197            .named_child(1)
198            .expect("No domain or expression found for letting statement");
199        match expr_or_domain.kind() {
200            "expression" => {
201                for name in temp_symbols {
202                    symbol_table.insert(Rc::new(Declaration::new_value_letting(
203                        Name::UserName(String::from(name)),
204                        parse_constraint(expr_or_domain, source_code),
205                    )));
206                }
207            }
208            "domain" => {
209                for name in temp_symbols {
210                    symbol_table.insert(Rc::new(Declaration::new_domain_letting(
211                        Name::UserName(String::from(name)),
212                        parse_domain(expr_or_domain, source_code),
213                    )));
214                }
215            }
216            _ => panic!("Unrecognized node in letting statement"),
217        }
218    }
219    symbol_table
220}

222fn parse_constraint(constraint: Node, source_code: &str) -> Expression {
223    match constraint.kind() {
224        "constraint" | "expression" => child_expr(constraint, source_code),
225        "not_expr" => Expression::Not(
226            Metadata::new(),
227            Box::new(child_expr(constraint, source_code)),
228        ),
229        "abs_value" => Expression::Abs(
230            Metadata::new(),
231            Box::new(child_expr(constraint, source_code)),
232        ),
233        "negative_expr" => Expression::Neg(
234            Metadata::new(),
235            Box::new(child_expr(constraint, source_code)),
236        ),
237        "exponent" | "product_expr" | "sum_expr" | "comparison" | "and_expr" | "or_expr"
238        | "implication" => {
239            let expr1 = child_expr(constraint, source_code);
240            let op = constraint.child(1).unwrap_or_else(|| {
241                panic!(
242                    "Error: missing node in expression of kind {}",
243                    constraint.kind()
244                )
245            });
246            let op_type = &source_code[op.start_byte()..op.end_byte()];
247            let expr2_node = constraint.child(2).unwrap_or_else(|| {
248                panic!(
249                    "Error: missing node in expression of kind {}",
250                    constraint.kind()
251                )
252            });
253            let expr2 = parse_constraint(expr2_node, source_code);
254
255            match op_type {
256                "**" => Expression::UnsafePow(Metadata::new(), Box::new(expr1), Box::new(expr2)),
257                "+" => Expression::Sum(Metadata::new(), vec![expr1, expr2]),
258                "-" => Expression::Minus(Metadata::new(), Box::new(expr1), Box::new(expr2)),
259                "*" => Expression::Product(Metadata::new(), vec![expr1, expr2]),
260                "/" => {
261                    //TODO: add checks for if division is safe or not
262                    Expression::UnsafeDiv(Metadata::new(), Box::new(expr1), Box::new(expr2))
263                }
264                "%" => {
265                    //TODO: add checks for if mod is safe or not
266                    Expression::UnsafeMod(Metadata::new(), Box::new(expr1), Box::new(expr2))
267                }
268                "=" => Expression::Eq(Metadata::new(), Box::new(expr1), Box::new(expr2)),
269                "!=" => Expression::Neq(Metadata::new(), Box::new(expr1), Box::new(expr2)),
270                "<=" => Expression::Leq(Metadata::new(), Box::new(expr1), Box::new(expr2)),
271                ">=" => Expression::Geq(Metadata::new(), Box::new(expr1), Box::new(expr2)),
272                "<" => Expression::Lt(Metadata::new(), Box::new(expr1), Box::new(expr2)),
273                ">" => Expression::Gt(Metadata::new(), Box::new(expr1), Box::new(expr2)),
274                "/\\" => Expression::And(Metadata::new(), vec![expr1, expr2]),
275                "\\/" => Expression::Or(Metadata::new(), vec![expr1, expr2]),
276                "->" => Expression::Imply(Metadata::new(), Box::new(expr1), Box::new(expr2)),
277                _ => panic!("Error: unsupported operator"),
278            }
279        }
280        "quantifier_expr" => {
281            let mut expr_list = Vec::new();
282            for expr in named_children(&constraint) {
283                expr_list.push(parse_constraint(expr, source_code));
284            }
285
286            let quantifier = constraint.child(0).unwrap_or_else(|| {
287                panic!(
288                    "Error: missing node in expression of kind {}",
289                    constraint.kind()
290                )
291            });
292            let quantifier_type = &source_code[quantifier.start_byte()..quantifier.end_byte()];
293
294            match quantifier_type {
295                "and" => Expression::And(Metadata::new(), expr_list),
296                "or" => Expression::Or(Metadata::new(), expr_list),
297                "min" => Expression::Min(Metadata::new(), expr_list),
298                "max" => Expression::Max(Metadata::new(), expr_list),
299                "sum" => Expression::Sum(Metadata::new(), expr_list),
300                "allDiff" => Expression::AllDiff(Metadata::new(), expr_list),
301                _ => panic!("Error: unsupported quantifier"),
302            }
303        }
304        "constant" => {
305            let child = constraint.child(0).unwrap_or_else(|| {
306                panic!(
307                    "Error: missing node in expression of kind {}",
308                    constraint.kind()
309                )
310            });
311            match child.kind() {
312                "integer" => {
313                    let constant_value = &source_code[child.start_byte()..child.end_byte()]
314                        .parse::<i32>()
315                        .unwrap();
316                    Expression::Atomic(
317                        Metadata::new(),
318                        Atom::Literal(Literal::Int(*constant_value)),
319                    )
320                }
321                "TRUE" => Expression::Atomic(Metadata::new(), Atom::Literal(Literal::Bool(true))),
322                "FALSE" => Expression::Atomic(Metadata::new(), Atom::Literal(Literal::Bool(false))),
323                _ => panic!("Error"),
324            }
325        }
326        "variable" => {
327            let variable_name =
328                String::from(&source_code[constraint.start_byte()..constraint.end_byte()]);
329            Expression::Atomic(
330                Metadata::new(),
331                Atom::Reference(Name::UserName(variable_name)),
332            )
333        }
334        _ => panic!("{} is not a recognized node kind", constraint.kind()),
335    }
336}

338fn named_children<'a>(node: &'a Node<'a>) -> impl Iterator<Item = Node<'a>> + 'a {
339    (0..node.named_child_count()).filter_map(|i| node.named_child(i))
340}

342fn child_expr(node: Node, source_code: &str) -> Expression {
343    let child = node
344        .named_child(0)
345        .unwrap_or_else(|| panic!("Error: missing node in expression of kind {}", node.kind()));
346    parse_constraint(child, source_code)
