23pub fn assert_eq_any_order<T: Eq + Hash + Debug + Clone>(a: &Vec<Vec<T>>, b: &Vec<Vec<T>>) {
24    assert_eq!(a.len(), b.len());
25
26    let mut a_rows: Vec<HashSet<T>> = Vec::new();
27    for row in a {
28        let hash_row = to_set(row);
29        a_rows.push(hash_row);
30    }
31
32    let mut b_rows: Vec<HashSet<T>> = Vec::new();
33    for row in b {
34        let hash_row = to_set(row);
35        b_rows.push(hash_row);
36    }
37
38    println!("{:?},{:?}", a_rows, b_rows);
39    for row in a_rows {
40        assert!(b_rows.contains(&row));
41    }
42}

44pub fn serialise_model(model: &ConjureModel) -> Result<String, JsonError> {
45    // A consistent sorting of the keys of json objects
46    // only required for the generated version
47    // since the expected version will already be sorted
48    let serde_model: SerdeModel = model.clone().into();
49    let generated_json = sort_json_object(&serde_json::to_value(serde_model)?, false);
50
51    // serialise to string
52    let generated_json_str = serde_json::to_string_pretty(&generated_json)?;
53
54    Ok(generated_json_str)
55}

115pub fn minion_solutions_from_json(
116    serialized: &str,
117) -> Result<Vec<HashMap<Name, Literal>>, anyhow::Error> {
118    let json: JsonValue = serde_json::from_str(serialized)?;
119
120    let json_array = json
121        .as_array()
122        .ok_or(Error::Parse("Invalid JSON".to_owned()))?;
123
124    let mut solutions = Vec::new();
125
126    for solution in json_array {
127        let mut sol = HashMap::new();
128        let solution = solution
129            .as_object()
130            .ok_or(Error::Parse("Invalid JSON".to_owned()))?;
131
132        for (var_name, constant) in solution {
133            let constant = match constant {
134                JsonValue::Number(n) => {
135                    let n = n
136                        .as_i64()
137                        .ok_or(Error::Parse("Invalid integer".to_owned()))?;
138                    Literal::Int(n as i32)
139                }
140                JsonValue::Bool(b) => Literal::Bool(*b),
141                _ => return Err(Error::Parse("Invalid constant".to_owned()).into()),
142            };
143
144            sol.insert(UserName(var_name.into()), constant);
145        }
146
147        solutions.push(sol);
148    }
149
150    Ok(solutions)
151}

192pub fn read_rule_trace(

217pub fn count_and_sort_rules(filename: &str) -> Result<JsonValue, anyhow::Error> {
218    let file_contents = read_to_string(filename)?;
219
220    let sorted_json_rules = if file_contents.trim().is_empty() {
221        let rule_count_message = json!({
222            "Number of rules applied": 0,
223        });
224        rule_count_message
225    } else {
226        let rule_count = file_contents.lines().count();
227        let mut sorted_json_rules = sort_json_rules(&file_contents)?;
228
229        let rule_count_message = json!({
230            "Number of rules applied": rule_count,
231        });
232
233        if let Some(array) = sorted_json_rules.as_array_mut() {
234            array.push(rule_count_message);
235        } else {
236            return Err(anyhow::anyhow!("Expected JSON array"));
237        }
238        sort_json_object(&sorted_json_rules, false)
239    };
240
241    let generated_sorted_json_rules = serde_json::to_string_pretty(&sorted_json_rules)?;
242
243    let mut file = OpenOptions::new()
244        .write(true)
245        .truncate(true)
246        .open(filename)?;
247
248    file.write_all(generated_sorted_json_rules.as_bytes())?;
249
250    Ok(sorted_json_rules)
251}

253fn sort_json_rules(json_rule_traces: &str) -> Result<JsonValue, anyhow::Error> {
254    let mut sorted_rule_traces = Vec::new();
255
256    for line in json_rule_traces.lines() {
257        let pretty_json = sort_json_object(&serde_json::from_str(line)?, true);
258        sorted_rule_traces.push(pretty_json);
259    }
260
261    Ok(JsonValue::Array(sorted_rule_traces))
262}
