18/// Parse an Essence file into a Model using the tree-sitter parser.
19pub fn parse_essence_file_native(
20    path: &str,
21    context: Arc<RwLock<Context<'static>>>,
22) -> Result<Model, EssenceParseError> {
23    let source_code = fs::read_to_string(path)
24        .unwrap_or_else(|_| panic!("Failed to read the source code file {path}"));
25    parse_essence_with_context(&source_code, context)
26}

28pub fn parse_essence_with_context(
29    src: &str,
30    context: Arc<RwLock<Context<'static>>>,
31) -> Result<Model, EssenceParseError> {
32    let (tree, source_code) = match get_tree(src) {
33        Some(tree) => tree,
34        None => {
35            return Err(EssenceParseError::TreeSitterError(
36                "Failed to parse source code".to_string(),
37            ));
38        }
39    };
40
41    let mut model = Model::new(context);
42    // let symbols = model.as_submodel().symbols().clone();
43    let root_node = tree.root_node();
44    for statement in named_children(&root_node) {
45        match statement.kind() {
46            "single_line_comment" => {}
47            "find_statement_list" => {
48                let var_hashmap = parse_find_statement(statement, &source_code);
49                for (name, domain) in var_hashmap {
50                    model
51                        .as_submodel_mut()
52                        .symbols_mut()
53                        .insert(DeclarationPtr::new_var(name, domain));
54                }
55            }
56            "constraint_list" => {
57                let mut constraint_vec: Vec<Expression> = Vec::new();
58                for constraint in named_children(&statement) {
59                    let current_symbols = model.as_submodel().symbols().clone();
60
61                    if constraint.kind() != "single_line_comment" {
62                        constraint_vec.push(parse_expression(
63                            constraint,
64                            &source_code,
65                            &statement,
66                            Some(&current_symbols),
67                        )?);
68                    }
69                }
70                model.as_submodel_mut().add_constraints(constraint_vec);
71            }
72            "language_label" => {}
73            "letting_statement_list" => {
74                let letting_vars = parse_letting_statement(statement, &source_code)?;
75                model.as_submodel_mut().symbols_mut().extend(letting_vars);
76            }
77            "dominance_relation" => {
78                let inner = statement
79                    .child(1)
80                    .expect("Expected a sub-expression inside `dominanceRelation`");
81                let current_symbols = model.as_submodel().symbols().clone();
82                let expr =
83                    parse_expression(inner, &source_code, &statement, Some(&current_symbols))?;
84                let dominance = Expression::DominanceRelation(Metadata::new(), Moo::new(expr));
85                if model.dominance.is_some() {
86                    return Err(EssenceParseError::syntax_error(
87                        "Duplicate dominance relation".to_string(),
88                        None,
89                    ));
90                }
91                model.dominance = Some(dominance);
92            }
93            _ => {
94                let kind = statement.kind();
95                return Err(EssenceParseError::syntax_error(
96                    format!("Unrecognized top level statement kind: {kind}"),
97                    None,
98                ));
99            }
100        }
101    }
102    Ok(model)
103}
